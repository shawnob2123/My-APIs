document.addEventListener('DOMContentLoaded', function () { document.getElementById('button').addEventListener('click', getJokes) function getJokes() { const url = 'https://official-joke-api.appspot.com/random_joke'; // xhr object const xhr = new XMLHttpRequest(); xhr.open('GET', url, true); xhr.onload = function () { if (this.status === 200 && this.readyState === 4) { const parsedResponseText = JSON.parse(this.responseText); setup = `
${parsedResponseText.setup}`; punchline = `
${parsedResponseText.punchline}`; document.getElementById('joke').innerHTML = setup; document.getElementById('punchline').innerHTML = punchline; console.log(parsedResponseText); } else if (this.status === 400) { console.log('Error..Try again'); } } xhr.send(); } })
